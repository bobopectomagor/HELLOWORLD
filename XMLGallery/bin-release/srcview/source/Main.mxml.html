<?xml version="1.0" encoding="utf-8"?>
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009"
               xmlns:s="library://ns.adobe.com/flex/spark"
               xmlns:mx="library://ns.adobe.com/flex/mx"
               minWidth="955" minHeight="600" applicationComplete="init()"
               viewSourceURL="srcview/index.html">
    <fx:Declarations>
        <!-- Place non-visual elements (e.g., services, value objects) here -->
        <fx:XML id="imageXML" source="assets/images.xml" />
    </fx:Declarations>
    <fx:Script>
        <![CDATA[
            import mx.controls.Image;
            public function init():void{
                for each(var imageData:XML in imageXML.image){
                    var myGalleryImage:GalleryImage = new GalleryImage(imageData.source.toString(),
                        imageData.description.toString());
                        grpImages.addElement(myGalleryImage);
                        
                }
                }
            
            public function filterGallery(imageType:String):void{
                
                    var filteredList:XMLList = imageXML.image.(@type==imageType);
                    grpImages.removeAllElements();
                    
                    for each(var imageData:XML in filteredList){
                        var myGalleryImage:GalleryImage = new GalleryImage(imageData.source.toString(),
                            imageData.description.toString());
                        grpImages.addElement(myGalleryImage);
                    }
            }
            
            public function all():void{
                grpImages.removeAllElements()
                    init();
            }
        ]]>
    </fx:Script>
    
    <s:HGroup x="66" y="57">
    </s:HGroup>
    
    <s:VGroup x="38" y="57">
    </s:VGroup>
    <s:Button x="10" y="28" label="Cool" click="filterGallery('cool')"/>
    <s:TileGroup id="grpImages" x="10" y="57">
        
    </s:TileGroup>
    <s:Button x="106" y="28" label="Scary" click="filterGallery('scary')"/>
    <s:Button x="210" y="28" label="Cute" click="filterGallery('cute')"/>
    <s:Button x="301" y="28" label="All" click="all()"/>
    
</s:Application>
