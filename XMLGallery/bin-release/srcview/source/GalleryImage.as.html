package
{
    import flash.events.MouseEvent;
    import spark.components.Group;
    import spark.components.Image;
    import spark.components.Label
    import com.greensock.TweenLite;
    public class GalleryImage extends Group
    {
        private var  _description:String;
        private var _source:String;    
        private var myLabel:Label;
        private var myImage:Image;
        
        public function GalleryImage(source:String, description:String):void
        {
            
            _source = source;
            _description = description;
            
            super();
            
            
            myImage = new Image();
            myImage.source = _source;
            myImage.alpha = .7;
            myImage.width = 200;
            myImage.height = 200;
            this.addElement(myImage);
            
            myLabel = new Label();
            myLabel.text = _description;
            myLabel.y = 210;
            myLabel.alpha = 0;
            this.addElement(myLabel);
            this.addEventListener(MouseEvent.MOUSE_OVER, onMover);
            this.addEventListener(MouseEvent.MOUSE_OUT, onMout);
        }
        
        private function onMover(e:MouseEvent):void{
            TweenLite.to(myLabel, .7,{alpha:1});
            TweenLite.to(myImage, .7,{alpha:1});
        }
        
        private function onMout(e:MouseEvent):void{
            TweenLite.to(myLabel, .7,{alpha:0});
            TweenLite.to(myImage, .7,{alpha:.7});
        }
    }
}
